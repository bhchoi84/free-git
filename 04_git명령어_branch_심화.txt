# git에 새로운 브랜치 생성 
git branch development
git branch stage
git branch master

# 로컬의 branch 확인 
git branch

# 다른 브랜치로 변경 
git checkout development

# Switched to branch 'development' 변경시 문구  노출 

# 작업 후 commit 후

# git checkout master 로 브랜치 변경 가능 
git checkout master

# 아침마다 확인 
# git branch 로 현재 브랜치 확인 
git branch 

# git pull origin master 
# git pull origin main 
# 항상 원격의 브랜치를 우선 받는다 ( 왜 ? 다른 작업자의 내용 추가를 위해 )
